<?php
/**
 * WP OAuth Error Class
 *
 * @todo Error descriptions are optional. Add a setting wether or not to include error description 
 * the admin area.
 */
class WO_Error extends WO_API
{

	public $errors;

	public function __construct( $args )
	{
		$this->set_errors();
		$error['error']=$args;

		// @todo Add setting in WP to set error description or not
		if (1===1)
			$error['error_description']=$this->errors[$args];

		// @todo Add a check here for a state when the class is added. The state is always required if it was presented in the original request
		$error['state'] = '123fsf';
		new WO_output($error);
	}

	/** 
	 * List if OAuth errors and thier respected error_descriptions
	 */
	private function set_errors ()
	{
		$errors = array(
			"invalid_request" => "The request is missing a required parameter, includes an invalid parameter value, includes a parameter more than once, or is otherwise malformed",
			"unauthorized_client" => "The client is not authorized to request an authorization code using this method",
			"access_denied" => "The resource owner or authorization server denied the request",
			"unsupported_response_type" => "The authorization server does not support obtaining an authorization code using this method",
			"invalid_scope" => "The requested scope is invalid, unknown, or malformed",
			"server_error" => "The authorization server encountered an unexpected condition that prevented it from fulfilling the request",
			"temporarily_unavailable" => "The authorization server is currently unable to handle the request due to a temporary overloading or maintenance of the server"
		);
		$this->errors = $errors;
	}
}
